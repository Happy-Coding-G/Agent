```sql
-- 建议：MySQL 8.0+
SET NAMES utf8mb4;
SET time_zone = '+00:00';

-- ========== A) users / spaces ==========

CREATE TABLE users (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_key      VARCHAR(64) NOT NULL UNIQUE,   -- 外部 IAM 的 user_id（字符串）
  display_name  VARCHAR(128) NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE user_auth (
  id              BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id         BIGINT UNSIGNED NOT NULL,
  identity_type   ENUM('password', 'phone', 'wechat', 'github') NOT NULL,
  identifier      VARCHAR(128) NOT NULL, -- 账号/手机号/OpenID
  credential      VARCHAR(255) NOT NULL, -- 密码哈希或 Token
  verified        TINYINT(1) DEFAULT 0,
  
  CONSTRAINT fk_auth_user FOREIGN KEY (user_id) REFERENCES users(id),
  UNIQUE KEY uk_identity (identity_type, identifier)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE spaces (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,   -- 前端 spaceId（建议：snowflake/uuid-short）
  name          VARCHAR(128) NOT NULL,
  owner_user_id BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  CONSTRAINT fk_spaces_owner FOREIGN KEY (owner_user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


-- ========== B) folders / files / versions / chunks ==========

-- 目录
CREATE TABLE folders (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    public_id VARCHAR(32) NOT NULL UNIQUE,
    space_id BIGINT UNSIGNED NOT NULL,
    parent_id BIGINT UNSIGNED NULL,  -- 父目录ID
    name VARCHAR(255) NOT NULL,
    path_cache VARCHAR(2048) NULL,   -- 全路径
    created_by BIGINT UNSIGNED NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP NULL,
    UNIQUE KEY uk_folder_sibling_name (space_id, parent_id, name)
) ENGINE=InnoDB;

-- 文件
CREATE TABLE files (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    public_id VARCHAR(32) NOT NULL UNIQUE,
    space_id BIGINT UNSIGNED NOT NULL,
    folder_id BIGINT UNSIGNED NOT NULL,
    name VARCHAR(255) NOT NULL,
    mime VARCHAR(128) NULL,
    size_bytes BIGINT UNSIGNED NULL,
    sha256 CHAR(64) NULL,
    status ENUM('active','archived','deleted') DEFAULT 'active',
    current_version_id BIGINT UNSIGNED NULL, -- 注意：稍后通过 ALTER 添加外键
    created_by BIGINT UNSIGNED NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 版本表：物理存储映射
CREATE TABLE file_versions (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    public_id VARCHAR(32) NOT NULL UNIQUE,
    file_id BIGINT UNSIGNED NOT NULL,
    version_no INT UNSIGNED NOT NULL,  -- 版本号
    object_key VARCHAR(512) NOT NULL,  -- 对象存储的 Key
    size_bytes BIGINT UNSIGNED NOT NULL,
    sha256 CHAR(64) NULL,
    created_by BIGINT UNSIGNED NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_file_versions_file FOREIGN KEY (file_id) REFERENCES files(id)
) ENGINE=InnoDB;

-- 上传表
CREATE TABLE uploads (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    public_id VARCHAR(32) NOT NULL UNIQUE,
    space_id BIGINT UNSIGNED NOT NULL,
    folder_id BIGINT UNSIGNED NOT NULL,
    filename VARCHAR(255) NOT NULL,
    size_bytes BIGINT UNSIGNED NOT NULL,
    status ENUM('init','uploading','completed','failed') DEFAULT 'init',
    created_by BIGINT UNSIGNED NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 循环外键
ALTER TABLE files ADD CONSTRAINT fk_files_current_version 
FOREIGN KEY (current_version_id) REFERENCES file_versions(id);


-- ========== D) chat / citations / tool_calls ==========

CREATE TABLE chat_sessions (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,      -- session_id
  space_id      BIGINT UNSIGNED NOT NULL,
  title         VARCHAR(255) NULL,
  created_by    BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  CONSTRAINT fk_chat_sessions_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  CONSTRAINT fk_chat_sessions_created_by FOREIGN KEY (created_by) REFERENCES users(id),
  KEY idx_chat_sessions_space (space_id, updated_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE chat_messages (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  session_id    BIGINT UNSIGNED NOT NULL,
  role          ENUM('user','assistant','system','tool') NOT NULL,
  content       MEDIUMTEXT NOT NULL,
  model         VARCHAR(128) NULL,
  token_count   INT UNSIGNED NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_chat_messages_session FOREIGN KEY (session_id) REFERENCES chat_sessions(id),
  KEY idx_chat_messages_session_time (session_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 每条 assistant message 的证据引用（chunk 级）
CREATE TABLE chat_citations (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  message_id    BIGINT UNSIGNED NOT NULL,
  chunk_id      BIGINT UNSIGNED NOT NULL,
  score         DOUBLE NULL,
  -- 可选：引用片段在 message 中的位置（便于高亮）
  span_start    INT UNSIGNED NULL,
  span_end      INT UNSIGNED NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_citations_message FOREIGN KEY (message_id) REFERENCES chat_messages(id),
  CONSTRAINT fk_citations_chunk FOREIGN KEY (chunk_id) REFERENCES file_chunks(id),
  KEY idx_citations_message (message_id),
  KEY idx_citations_chunk (chunk_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Tool call 记录：供 Output 面板、审计回放、debug
CREATE TABLE tool_calls (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  message_id    BIGINT UNSIGNED NULL,             -- 关联到某条 message（可选）
  session_id    BIGINT UNSIGNED NULL,
  tool_name     VARCHAR(128) NOT NULL,
  tool_args     JSON NULL,
  tool_result   JSON NULL,
  status        ENUM('ok','error','timeout') NOT NULL DEFAULT 'ok',
  started_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  ended_at      TIMESTAMP NULL,

  CONSTRAINT fk_tool_calls_message FOREIGN KEY (message_id) REFERENCES chat_messages(id),
  CONSTRAINT fk_tool_calls_session FOREIGN KEY (session_id) REFERENCES chat_sessions(id),
  KEY idx_tool_calls_session_time (session_id, started_at),
  KEY idx_tool_calls_tool (tool_name, started_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


-- ========== E) markdown docs / versions ==========

CREATE TABLE markdown_docs (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,      -- doc_id
  space_id      BIGINT UNSIGNED NOT NULL,
  -- 可选：让 markdown 也能作为 file 的一种类型（双向关联）
  file_id       BIGINT UNSIGNED NULL,
  title         VARCHAR(255) NULL,
  current_version_id BIGINT UNSIGNED NULL,
  created_by    BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  CONSTRAINT fk_md_docs_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  CONSTRAINT fk_md_docs_file FOREIGN KEY (file_id) REFERENCES files(id),
  CONSTRAINT fk_md_docs_created_by FOREIGN KEY (created_by) REFERENCES users(id),
  KEY idx_md_docs_space (space_id, updated_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE markdown_versions (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  doc_id        BIGINT UNSIGNED NOT NULL,
  version_no    INT UNSIGNED NOT NULL,
  content       MEDIUMTEXT NOT NULL,
  patch_type    ENUM('json_patch','unified_diff','full') NOT NULL DEFAULT 'full',
  patch         MEDIUMTEXT NULL,
  base_version_id BIGINT UNSIGNED NULL,
  created_by    BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_md_versions_doc FOREIGN KEY (doc_id) REFERENCES markdown_docs(id),
  CONSTRAINT fk_md_versions_base FOREIGN KEY (base_version_id) REFERENCES markdown_versions(id),
  CONSTRAINT fk_md_versions_created_by FOREIGN KEY (created_by) REFERENCES users(id),
  UNIQUE KEY uk_md_doc_version (doc_id, version_no),
  KEY idx_md_versions_doc_time (doc_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

ALTER TABLE markdown_docs
  ADD CONSTRAINT fk_md_docs_current_version FOREIGN KEY (current_version_id) REFERENCES markdown_versions(id);


-- ========== F) knowledge graph (MySQL-native) ==========

CREATE TABLE kg_versions (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  space_id      BIGINT UNSIGNED NOT NULL,
  version_no    INT UNSIGNED NOT NULL,
  note          VARCHAR(255) NULL,
  created_by    BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_kg_versions_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  CONSTRAINT fk_kg_versions_created_by FOREIGN KEY (created_by) REFERENCES users(id),
  UNIQUE KEY uk_kg_space_version (space_id, version_no),
  KEY idx_kg_versions_space_time (space_id, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE kg_nodes (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  version_id    BIGINT UNSIGNED NOT NULL,
  node_key      VARCHAR(255) NOT NULL,            -- 稳定键（URI/ID/名称hash）
  label         VARCHAR(255) NULL,
  node_type     VARCHAR(64) NULL,
  props         JSON NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_kg_nodes_version FOREIGN KEY (version_id) REFERENCES kg_versions(id),
  UNIQUE KEY uk_kg_node_unique (version_id, node_key),
  KEY idx_kg_nodes_version (version_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE kg_edges (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  version_id    BIGINT UNSIGNED NOT NULL,
  src_node_id   BIGINT UNSIGNED NOT NULL,
  dst_node_id   BIGINT UNSIGNED NOT NULL,
  rel_type      VARCHAR(64) NOT NULL,
  props         JSON NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_kg_edges_version FOREIGN KEY (version_id) REFERENCES kg_versions(id),
  CONSTRAINT fk_kg_edges_src FOREIGN KEY (src_node_id) REFERENCES kg_nodes(id),
  CONSTRAINT fk_kg_edges_dst FOREIGN KEY (dst_node_id) REFERENCES kg_nodes(id),
  KEY idx_kg_edges_version (version_id),
  KEY idx_kg_edges_src (src_node_id),
  KEY idx_kg_edges_dst (dst_node_id),
  KEY idx_kg_edges_rel (version_id, rel_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- provenance：节点/边来自哪个 chunk（可多条证据）
CREATE TABLE kg_evidence (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  version_id    BIGINT UNSIGNED NOT NULL,
  target_type   ENUM('node','edge') NOT NULL,
  target_id     BIGINT UNSIGNED NOT NULL,         -- 指向 kg_nodes.id 或 kg_edges.id
  chunk_id      BIGINT UNSIGNED NOT NULL,
  extractor     VARCHAR(128) NULL,                -- rule/llm/ner/...
  model         VARCHAR(128) NULL,
  confidence    DOUBLE NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_kg_evidence_version FOREIGN KEY (version_id) REFERENCES kg_versions(id),
  CONSTRAINT fk_kg_evidence_chunk FOREIGN KEY (chunk_id) REFERENCES file_chunks(id),
  KEY idx_kg_evidence_target (target_type, target_id),
  KEY idx_kg_evidence_chunk (chunk_id),
  KEY idx_kg_evidence_version (version_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


-- ========== G) assets / links / tags ==========

CREATE TABLE assets (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  public_id     VARCHAR(32) NOT NULL UNIQUE,
  space_id      BIGINT UNSIGNED NOT NULL,
  name          VARCHAR(255) NOT NULL,
  asset_type    VARCHAR(64) NOT NULL,             -- doc/metric/kg_domain/knowledge_card/...
  security_level ENUM('public','internal','confidential','secret') NOT NULL DEFAULT 'internal',
  status        ENUM('active','archived','deleted') NOT NULL DEFAULT 'active',
  description   TEXT NULL,
  props         JSON NULL,
  created_by    BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  CONSTRAINT fk_assets_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  CONSTRAINT fk_assets_created_by FOREIGN KEY (created_by) REFERENCES users(id),
  KEY idx_assets_space_type (space_id, asset_type),
  KEY idx_assets_space_updated (space_id, updated_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 资产与任意资源的关联（来源/派生/引用/归档等）
CREATE TABLE asset_links (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  asset_id      BIGINT UNSIGNED NOT NULL,
  resource_type ENUM('space','folder','file','file_version','file_chunk','markdown_doc','markdown_version','kg_node','kg_edge','chat_session','chat_message') NOT NULL,
  resource_id   BIGINT UNSIGNED NOT NULL,
  link_role     VARCHAR(64) NOT NULL DEFAULT 'source',   -- source/derived/reference/owner/...
  props         JSON NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_asset_links_asset FOREIGN KEY (asset_id) REFERENCES assets(id),
  KEY idx_asset_links_asset (asset_id),
  KEY idx_asset_links_resource (resource_type, resource_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 通用标签（可用于文件/目录/资产/图谱等）
CREATE TABLE tags (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  space_id      BIGINT UNSIGNED NOT NULL,
  name          VARCHAR(64) NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_tags_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  UNIQUE KEY uk_tag_space_name (space_id, name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE resource_tags (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  tag_id        BIGINT UNSIGNED NOT NULL,
  resource_type ENUM('folder','file','asset','markdown_doc','kg_node','kg_edge') NOT NULL,
  resource_id   BIGINT UNSIGNED NOT NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_resource_tags_tag FOREIGN KEY (tag_id) REFERENCES tags(id),
  UNIQUE KEY uk_tag_resource (tag_id, resource_type, resource_id),
  KEY idx_resource_tags_resource (resource_type, resource_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


-- ========== H) ACL / audit events ==========

-- 资源级 ACL：可用于 folder/file/asset/chat/kg/md 等
CREATE TABLE acl_entries (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  space_id      BIGINT UNSIGNED NOT NULL,
  resource_type ENUM('space','folder','file','asset','markdown_doc','chat_session','kg_version') NOT NULL,
  resource_id   BIGINT UNSIGNED NOT NULL,
  principal_type ENUM('user','group') NOT NULL DEFAULT 'user',
  principal_key VARCHAR(64) NOT NULL,             -- user.user_key 或 group_key
  -- 权限位：建议用 bitmask，例如：1=read,2=write,4=delete,8=share,16=admin
  perm_mask     INT UNSIGNED NOT NULL,
  inherited     TINYINT(1) NOT NULL DEFAULT 0,    -- 是否继承（用于目录向下继承）
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_acl_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  KEY idx_acl_resource (resource_type, resource_id),
  KEY idx_acl_principal (principal_type, principal_key),
  KEY idx_acl_space (space_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 审计日志：每次敏感操作都落这里（查/读/写/删/分享/导出/工具调用等）
CREATE TABLE audit_events (
  id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  space_id      BIGINT UNSIGNED NOT NULL,
  actor_user_key VARCHAR(64) NOT NULL,
  action        VARCHAR(64) NOT NULL,             -- e.g. folder.create / file.read / md.patch / kg.query / chat.send
  resource_type VARCHAR(32) NOT NULL,
  resource_id   BIGINT UNSIGNED NULL,
  metadata      JSON NULL,
  ip            VARCHAR(64) NULL,
  user_agent    VARCHAR(255) NULL,
  created_at    TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_audit_space FOREIGN KEY (space_id) REFERENCES spaces(id),
  KEY idx_audit_space_time (space_id, created_at),
  KEY idx_audit_actor_time (actor_user_key, created_at),
  KEY idx_audit_action_time (action, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

```
