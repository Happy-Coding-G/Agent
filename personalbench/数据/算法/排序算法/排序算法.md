### 冒泡排序

```
方法：从头开始两两比较，将大的数放置到后面，每一轮确定一个最大的数
时间复杂度：o（n2）
空间复杂度：o（1）
```

```
void buubbleSort(int arr[], int n){
    int i, j, temp;
    for(i=n-1;i>0;i--)
    {
        for(j=0;j<i;j++)
        {
            if(arr[j]  > arr[j+1])
            {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}
```

### 选择排序

```
方法：从未排序的里面选出最小的放到排好序的数组的最后面
时间复杂度：o（n2）
空间复杂度：o（1）
```

```
void selectionSort(int arr[], int n){
    int i, j, temp;
    for(i=0;i<n-1;i++)
    {
        int minIndex = i;
        for(j=i;j<n-1;j++)
        {
            if(arr[minIndex] > arr[j])
            {
                minIndex = j;
            }
        }
        temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }
}
```

### 插入排序

```
方法：将未排序的数组从最后面插入到已排好序的数组
时间复杂度：o（n2）
空间复杂度：o（1）
```

```
void insert_sort(int arr[], int n){
    int i, j, temp;
    for(i=1;i<n;i++)
    {
        for(j=i-1;j>=0;j--)
        {
            if(arr[i] < arr[j])
            {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}
```

### 归并排序

```
方法：分-和，将数组递归地分为两组，合并有序的两个数组
时间复杂度：o（nlogn）
空间复杂度：o（n）
```

```
void merge(vector<int>& arr, int left, int mid, int right){
    int n1 = mid - left + 1;
    int n2 = right - mid;

    vector<int> L(n1);
    vector<int> R(n2);

    for(int i=0;i<n1;i++)
    {
        L[i] = arr[left + i];
    }
    for(int i=0;i<n2;i++)
    {
        R[i] = arr[mid + i + 1];
    }

    int i=0, j=0, index=left;
    while(i<n1 && j<=n2)
    {
        if(L[i] <= R[j])
        {
            arr[index] = L[i];
            i++;
            index++;
        }
        else
        {
            arr[index] = R[j];
            j++;
            index++;
        }
    }

    while(i<n1)
    {
        arr[index] = L[i];
        i++;
        index++;
    }
    while(j<n2)
    {
        arr[index] = R[j];
        j++;
        index++;
    }
}

void merge_sort(vector<int>& arr, int left, int right){
    if(left < right)
    {
        int mid = left + (right-left)/2;

        merge_sort(arr, left, mid);
        merge_sort(arr, mid+1, right);

        merge(arr, left, mid, right);
    }

}
```
